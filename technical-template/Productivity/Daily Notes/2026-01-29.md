```js-engine
const mb = engine.getPlugin('obsidian-meta-bind-plugin').api;

const comp = new obsidian.Component(component);

const bWeights = mb.parseBindTarget('weights', context.file.path);
const bCardio = mb.parseBindTarget('cardio', context.file.path);
const bGuitar = mb.parseBindTarget('guitar', context.file.path);
const bPiano = mb.parseBindTarget('piano', context.file.path);
const bSinging = mb.parseBindTarget('singing', context.file.path);
const bRecording = mb.parseBindTarget('recording', context.file.path);
const bDrawing = mb.parseBindTarget('drawing', context.file.path);
const bModelling = mb.parseBindTarget('modelling', context.file.path);
const bCoding = mb.parseBindTarget('coding', context.file.path);
const bModing = mb.parseBindTarget('moding', context.file.path);
const bElectronics = mb.parseBindTarget('electronics', context.file.path);
const bPrinting = mb.parseBindTarget('printing', context.file.path);
const bBookReading = mb.parseBindTarget('book_reading', context.file.path);
const bArticleReading = mb.parseBindTarget('article_reading', context.file.path);
const bObsidian = mb.parseBindTarget('obsidian', context.file.path);
const bVideoGames = mb.parseBindTarget('videogames', context.file.path);
const bMoviesTV = mb.parseBindTarget('moviestv', context.file.path);
const bCV = mb.parseBindTarget('cv', context.file.path);
const bttrpg = mb.parseBindTarget('ttrpg', context.file.path);

function render() {
	comp.unload()
	comp.load()
	container.empty();

    const targets = [
        { obj: bWeights, flag: "weights", time: "weightstime", name: "weights"},
        { obj: bCardio, flag: "cardio", time: "cardiotime", name: "cardio"},
        { obj: bGuitar, flag: "guitar", time: "guitartime", name: "guitar"},
        { obj: bPiano, flag: "piano", time: "pianotime", name: "piano"},
        { obj: bSinging, flag: "singing", time: "singingtime", name: "singing"},
        { obj: bRecording, flag: "recording", time: "recordingtime", name: "recording"},
        { obj: bDrawing, flag: "drawing", time: "drawingtime", name: "drawing"},
        { obj: bModelling, flag: "modelling", time: "modellingtime", name: "modelling"},
        { obj: bCoding, flag: "coding", time: "codingtime", name: "coding"},
        { obj: bModing, flag: "moding", time: "modingtime", name: "moding"},
        { obj: bElectronics, flag: "electronics", time: "makertime", name: "electronics"},
        { obj: bPrinting, flag: "printing", time: "makertime", name: "printing"},
        { obj: bBookReading, flag: "book_reading", time: "readingtime", name: "book reading"},
        { obj: bArticleReading, flag: "article_reading", time: "readingtime", name: "article reading"},
        { obj: bObsidian, flag: "obsidian", time: "obsidiantime", name: "obsidian"},
        { obj: bVideoGames, flag: "videogames", time: "videogametime", name: "video games"},
        { obj: bMoviesTV, flag: "moviestv", time: "moviestvtime", name: "movies and TV"},
        { obj: bCV, flag: "cv", time: "cvtime", name: "CV"},
        { obj: bttrpg, flag: "ttrpg", time: "ttrpgtime", name: "ttrpg"}
    ];

    targets.forEach(target => {
        if (mb.getMetadata(target.obj)) {
            // Add Slider
            const divcontainer = document.createElement('div');
            container.appendChild(divcontainer);

            const sliderContainer = mb.createInlineFieldFromString(`INPUT[slider(addLabels, title(${target.flag}), minValue(0), maxValue(360), stepSize(30), showcase):${target.time}]`, context.file.path, undefined);

            mb.wrapInMDRC(sliderContainer, divcontainer, comp);

            // Add label and slider output
            const divcontainer2 = document.createElement('div');
            container.appendChild(divcontainer2);

            const timeContainer = mb.createInlineFieldFromString(`VIEW[${target.flag}\: {${target.time}}][text]`, context.file.path, undefined);

            mb.wrapInMDRC(timeContainer, divcontainer2, comp);
        }
    });
}

reactive = engine.reactive(render);

const bindTargets = [
    bWeights,
    bCardio,
    bGuitar,
    bPiano,
    bSinging,
    bRecording,
    bDrawing,
    bModelling,
    bCoding,
    bModing,
    bElectronics,
    bPrinting,
    bBookReading,
    bArticleReading,
    bObsidian,
    bVideoGames,
    bMoviesTV,
    bCV,
    bttrpg
];

bindTargets.forEach((bindTarget) => {
    mb.subscribeToMetadata(bindTarget, component, (value) =>
        reactive.refresh(value)
    );
});

return reactive;
```
